46  template<I2CMode MODE_, bool HAS_STATUS_ = false, typename STATUS_HOOK_ = I2C_STATUS_HOOK>
47  class ATtinyI2CSyncHandler
48  {
49  private:
50  using MODE_TRAIT = I2CMode_trait<MODE_>;
51  using I2C_TRAIT = board_traits::TWI_trait;
52  using REG8 = board_traits::REG8;
53  using STATUS = I2CStatusSupport<HAS_STATUS_, STATUS_HOOK_>;
54 
55  public:
56  ATtinyI2CSyncHandler(STATUS_HOOK_ status_hook = nullptr) : status_hook_{status_hook}
57  {
58  // set SDA/SCL default directions
59  I2C_TRAIT::PORT |= I2C_TRAIT::SCL_SDA_MASK;
60  I2C_TRAIT::DDR |= I2C_TRAIT::SCL_SDA_MASK;
61  }

63  void begin_()
64  {
65  // 1. Force 1 to data
66  USIDR_ = UINT8_MAX;
67  // 2. Enable TWI
68  // Set USI I2C mode, enable software clock strobe (USITC)
69  USICR_ = bits::BV8(USIWM1, USICS1, USICLK);
70  // Clear all interrupt flags
71  USISR_ = bits::BV8(USISIF, USIOIF, USIPF, USIDC);
72  // 3. Set SDA as output
73  SDA_OUTPUT();
74  }
75 
76  void end_()
77  {
78  // Disable TWI
79  USICR_ = 0;
80  // Set SDA back to INPUT
81  SDA_INPUT();
82  }

84  // Low-level methods to handle the bus in an asynchronous way
85  bool exec_start_()
86  {
87  return notify_status(send_start_(),
88  Status::START_TRANSMITTED, Status::ARBITRATION_LOST);
89  }

91  bool exec_repeat_start_()
92  {
93  return notify_status(send_start_(),
94  Status::REPEAT_START_TRANSMITTED, Status::ARBITRATION_LOST);
95  }

98  {

115  bool exec_receive_data_(bool last_byte, uint8_t& data)
116  {
117  SDA_INPUT();
118  data = transfer(USISR_DATA);
119  // Send ACK (or NACK if last byte)
120  USIDR_ = (last_byte ? UINT8_MAX : 0x00);
121  transfer(USISR_ACK);
122  //TODO return ((transfer(USISR_ACK) & 0x01U) == 0);
123  return notify_status(true, Status::DATA_RECEIVED_ACK, Status::DATA_RECEIVED_NACK);
124  }

126  void exec_stop_()
127  {
128  // Pull SDA low
129  SDA_LOW();
130  // Release SCL
131  SCL_HIGH();
132  _delay_loop_1(MODE_TRAIT::T_SU_STO);
133  // Release SDA
134  SDA_HIGH();
135  _delay_loop_1(MODE_TRAIT::T_BUF);
136  }
137 
138  private:
139  static constexpr const REG8 USIDR_{USIDR};
140  static constexpr const REG8 USISR_{USISR};
141  static constexpr const REG8 USICR_{USICR};
142 
143  // Constant values for USISR
144  // For byte transfer, we set counter to 0 (16 ticks => 8 clock cycles)
145  static constexpr const uint8_t USISR_DATA = bits::BV8(USISIF, USIOIF, USIPF, USIDC);
146  // For acknowledge bit, we start counter at 0E (2 ticks: 1 raising and 1 falling edge)
147  static constexpr const uint8_t USISR_ACK = USISR_DATA | (0x0E << USICNT0);
148 
149  bool notify_status(bool as_expected, uint8_t good, uint8_t bad)
150  {
151  status_hook_.call_hook(good, (as_expected ? good : bad));
152  return as_expected;
153  }
154 
155  void SCL_HIGH()
156  {
157  I2C_TRAIT::PORT |= bits::BV8(I2C_TRAIT::BIT_SCL);
158  I2C_TRAIT::PIN.loop_until_bit_set(I2C_TRAIT::BIT_SCL);
159  }
160 
161  void SCL_LOW()
162  {
163  I2C_TRAIT::PORT &= bits::CBV8(I2C_TRAIT::BIT_SCL);
164  }
165 
166  void SDA_HIGH()
167  {
168  I2C_TRAIT::PORT |= bits::BV8(I2C_TRAIT::BIT_SDA);
169  }
170 
171  void SDA_LOW()
172  {
173  I2C_TRAIT::PORT &= bits::CBV8(I2C_TRAIT::BIT_SDA);
174  }
175 
176  void SDA_INPUT()
177  {
178  I2C_TRAIT::DDR &= bits::CBV8(I2C_TRAIT::BIT_SDA);
179  }
180 
181  void SDA_OUTPUT()
182  {
183  I2C_TRAIT::DDR |= bits::BV8(I2C_TRAIT::BIT_SDA);
184  }

186  bool send_start_()
187  {
188  // Ensure SCL is HIGH
189  SCL_HIGH();
190  // Wait for Tsu-sta
191  _delay_loop_1(MODE_TRAIT::T_SU_STA);
192  // Now we can generate start condition
193  // Force SDA low for Thd-sta
194  SDA_LOW();
195  _delay_loop_1(MODE_TRAIT::T_HD_STA);
196  // Pull SCL low
197  SCL_LOW();
198  // Release SDA (force high)
199  SDA_HIGH();
200  // Check START transmission with USISIF flag
201  return USISR_ & bits::BV8(USISIF);
202  }
203 
204  bool send_byte_impl(uint8_t data)
205  {
206  // Set SCL low
207  SCL_LOW();
208  // Transfer address byte
209  USIDR_ = data;
210  transfer(USISR_DATA);
211  // For acknowledge, first set SDA as input
212  SDA_INPUT();
213  return ((transfer(USISR_ACK) & 0x01U) == 0);
214  }
215 
216  uint8_t transfer(uint8_t USISR_count)
217  {
218  //Rework according to AVR310
219  // Init counter (8 bits or 1 bit for acknowledge)
220  USISR_ = USISR_count;
221  do
222  {
223  _delay_loop_1(MODE_TRAIT::T_LOW);
224  // clock strobe (SCL raising edge)
225  USICR_ = bits::BV8(USIWM1, USICS1, USICLK, USITC);
226  I2C_TRAIT::PIN.loop_until_bit_set(I2C_TRAIT::BIT_SCL);
227  _delay_loop_1(MODE_TRAIT::T_HIGH);
228  // clock strobe (SCL falling edge)
229  USICR_ = bits::BV8(USIWM1, USICS1, USICLK, USITC);
230  }
231  while ((USISR_ & bits::BV8(USIOIF)) == 0);
232  _delay_loop_1(MODE_TRAIT::T_LOW);
233  // Read data
234  uint8_t data = USIDR_;
235  USIDR_ = UINT8_MAX;
236  // Release SDA
237  SDA_OUTPUT();
238  return data;
239  }

242  };
