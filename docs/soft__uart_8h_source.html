116  void compute_times(uint32_t rate, StopBits stop_bits)
117  {
118  // Calculate timing for TX in number of cycles
119  uint16_t bit_time = uint16_t(F_CPU / rate);
120  // 11 or 12 cycles + delay counted from start bit (cbi) to first bit (sbi or cbi)
121  start_bit_tx_time_ = (bit_time - 12) / 4;
122  // 11 or 12 cycles + delay counted from first bit (sbi or cbi) to second bit (sbi or cbi)
123  interbit_tx_time_ = (bit_time - 12) / 4;
124  // For stop bit we lengthten the bit duration of 25% to guarantee alignment of RX side on stop duration
125  stop_bit_tx_time_ = (bit_time / 4) * 5 / 4;
126  if (stop_bits == StopBits::TWO) stop_bit_tx_time_ *= 2;
127  }
128 
129  static Parity calculate_parity(Parity parity, uint8_t value)
130  {
131  if (parity == Parity::NONE) return Parity::NONE;
132  bool odd = false;
133  while (value)
134  {
135  if (value & 0x01) odd = !odd;
136  value >>= 1;
137  }
138  return (odd ? Parity::ODD : Parity::EVEN);
139  }

167  template<board::DigitalPin DPIN> void AbstractUATX::write_(Parity parity, uint8_t value)
168  {
169  using TX = gpio::FastPinType<DPIN>;
170  // Pre-calculate all what we need: parity bit
171  Parity parity_bit = calculate_parity(parity, value);
172 
173  // Write start bit
174  TX::clear();
175  // Wait before starting actual value transmission
176  _delay_loop_2(start_bit_tx_time_);
177  for (uint8_t bit = 0; bit < 8; ++bit)
178  {
179  if (value & 0x01)
180  TX::set();
181  else
182  TX::clear();
183  value >>= 1;
184  _delay_loop_2(interbit_tx_time_);
185  }
186  // Add parity if needed
187  if (parity_bit != Parity::NONE)
188  {
189  if (parity_bit == parity)
190  TX::clear();
191  else
192  TX::set();
193  _delay_loop_2(interbit_tx_time_);
194  }
195  // Add stop bit
196  TX::set();
197  _delay_loop_2(stop_bit_tx_time_);
198  }

291  void compute_times(uint32_t rate, UNUSED bool has_parity, UNUSED StopBits stop_bits)
292  {
293  // Calculate timing for RX in number of cycles
294  uint16_t bit_time = uint16_t(F_CPU / rate);
295 
296  // Actual timing is based on number of times to count 4 cycles, because we use _delay_loop_2()
297 
298  // Time to wait (_delay_loop_2) between detection of start bit and sampling of first bit
299  // For sampling of first bit we wait until middle of first bit
300  // We remove processing time due to ISR call and ISR code:
301  // - 3 cycles to generate the PCI interrupt
302  // - 1-4 (take 2) cycles to complete current instruction
303  // - 4 cycles to process the interrupt + 2 cycles rjmp in vector table
304  // - 32 cycles spent in PCINT vector to save context and check stop bit (sbic)
305  // - 8 cycles to setup stack variables
306  // - (4N) + 4 in delay
307  // - 8 cycles until first bit sample read (sbis)
308  start_bit_rx_time_ = compute_delay(3 * bit_time / 2, 3 + 2 + 4 + 2 + 32 + 8 + 4 + 8);
309 
310  // Time to wait (_delay_loop_2) between sampling of 2 consecutive data bits
311  // This is also use between last bit and parity bit (if checked) or stop bit
312  // We have to wait exactly for `bit_time` cycles
313  // We remove processing time due to each bit sampling and data value update
314  // - 10+4N cycles elapse between processing of each bit
315  interbit_rx_time_ = compute_delay(bit_time, 10);
316 
317  // No extra delay is used after sampling of first stop bit (as there are already
318  // enough code cycles used for pushing data and restoring context on ISR leave)
319  // Actually, >80 cycles elapse until next PCI can be handled
320  }

327  // We add 3 cycles to allow rounding
328  return (total_cycles > less_cycles) ? ((total_cycles - less_cycles + 3) / 4) : 1;

339  template<board::DigitalPin DPIN> void AbstractUARX::pin_change(Parity parity, Errors& errors)
340  {
341  using RX = gpio::FastPinType<DPIN>;
342  // Check RX is low (start bit)
343  if (RX::value()) return;
344  uint8_t value = 0;
345  bool odd = false;
346  errors.has_errors = 0;
347  // Wait for start bit to finish
348  _delay_loop_2(start_bit_rx_time_);
349  // Read first 7 bits
350  for (uint8_t i = 0; i < 8; ++i)
351  {
352  if (RX::value())
353  {
354  value |= 0x80;
355  odd = !odd;
356  }
357  if (i < 7)
358  value >>= 1;
359  _delay_loop_2(interbit_rx_time_);
360  }
361 
362  if (parity != Parity::NONE)
363  {
364  // Check parity bit
365  bool parity_bit = (parity == Parity::ODD ? !odd : odd);
366  // Check parity bit
367  errors.parity_error = (RX::value() != parity_bit);
368  _delay_loop_2(interbit_rx_time_);
369  }
370 
371  // Check we receive a stop bit
372  if (!RX::value())
373  errors.frame_error = true;
374  // Push value if no error
375  if (errors.has_errors == 0)
376  errors.queue_overflow = !in_().queue().push_(char(value));
377  }
