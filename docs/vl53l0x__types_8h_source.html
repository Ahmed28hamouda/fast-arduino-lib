93  class TimeoutUtilities
94  {
95  private:
96  static constexpr uint32_t PLL_PERIOD_PS = 1655UL;
97  static constexpr uint32_t MACRO_PERIOD_VCLKS = 2304UL;
98 
99  public:
100  // timeout in macro periods must be encoded on 16 bits, as (LSB.2^MSB)+1
101  static constexpr uint16_t encode_timeout(uint32_t timeout_macro_clks)
102  {
103  constexpr uint32_t MSB24_MASK = 0xFF'FF'FF'00UL;
104  constexpr uint32_t LSB8_MASK = 0xFFUL;
105 
106  if (timeout_macro_clks == 0UL) return 0U;
107  uint32_t lsb = timeout_macro_clks - 1UL;
108  uint16_t msb = 0;
109  while (lsb & MSB24_MASK)
110  {
111  lsb >>= 1;
112  ++msb;
113  }
114  return utils::as_uint16_t(uint8_t(msb), uint8_t(lsb & LSB8_MASK));
115  }
116 
117  // timeout in macro periods is encoded on 16 bits, as (LSB.2^MSB)+1
118  static constexpr uint32_t decode_timeout(uint16_t encoded_timeout)
119  {
120  const uint32_t lsb = utils::low_byte(encoded_timeout);
121  const uint32_t msb = utils::high_byte(encoded_timeout);
122  return (lsb << msb) + 1UL;
123  }

125  static constexpr uint32_t calculate_macro_period_ps(uint8_t vcsel_period_pclks)
126  {
127  constexpr uint32_t PS_TO_NS_DIVIDER = 1000UL;
128  constexpr uint32_t PS_TO_NS_ROUNDING_ADDITION = 500UL;
129  return ((PLL_PERIOD_PS * MACRO_PERIOD_VCLKS * vcsel_period_pclks) + PS_TO_NS_ROUNDING_ADDITION)
130  / PS_TO_NS_DIVIDER;
131  }
132  static constexpr uint32_t calculate_timeout_us(uint16_t timeout_period_mclks, uint8_t vcsel_period_pclks)
133  {
134  constexpr uint32_t NS_TO_US_DIVIDER = 1000UL;
135  constexpr uint32_t NS_TO_US_ROUNDING_ADDITION = 500UL;
136  const uint32_t macro_period_ns = calculate_macro_period_ps(vcsel_period_pclks);
137  return ((timeout_period_mclks * macro_period_ns) + NS_TO_US_ROUNDING_ADDITION) / NS_TO_US_DIVIDER;
138  }
139  static constexpr uint32_t calculate_timeout_mclks(uint16_t timeout_period_us, uint8_t vcsel_period_pclks)
140  {
141  constexpr uint32_t US_TO_NS_MULTIPLIER = 1000UL;
142  const uint32_t macro_period_ns = calculate_macro_period_ps(vcsel_period_pclks);
143  return ((timeout_period_us * US_TO_NS_MULTIPLIER) + (macro_period_ns / 2)) / macro_period_ns;
144  }
