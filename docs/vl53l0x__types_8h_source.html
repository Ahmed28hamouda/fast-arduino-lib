93  class TimeoutUtilities
94  {
95  private:
96  static constexpr uint32_t PLL_PERIOD_PS = 1655UL;
97  static constexpr uint32_t MACRO_PERIOD_VCLKS = 2304UL;
98 
99  public:
100  // timeout in macro periods must be encoded on 16 bits, as (LSB.2^MSB)+1
101  static constexpr uint16_t encode_timeout(uint32_t timeout_macro_clks)
102  {
103  if (timeout_macro_clks == 0UL) return 0U;
104  uint32_t lsb = timeout_macro_clks - 1UL;
105  uint16_t msb = 0;
106  while (lsb & 0xFFFFFF00UL)
107  {
108  lsb >>= 1;
109  ++msb;
110  }
111  return utils::as_uint16_t(uint8_t(msb), uint8_t(lsb & 0xFFUL));
112  }
113 
114  // timeout in macro periods is encoded on 16 bits, as (LSB.2^MSB)+1
115  static constexpr uint32_t decode_timeout(uint16_t encoded_timeout)
116  {
117  const uint32_t lsb = utils::low_byte(encoded_timeout);
118  const uint32_t msb = utils::high_byte(encoded_timeout);
119  return (lsb << msb) + 1UL;
120  }
121 
122  static constexpr uint32_t calculate_macro_period_ps(uint8_t vcsel_period_pclks)
123  {
124  return ((PLL_PERIOD_PS * MACRO_PERIOD_VCLKS * vcsel_period_pclks) + 500UL) / 1000UL;
125  }
126  static constexpr uint32_t calculate_timeout_us(uint16_t timeout_period_mclks, uint8_t vcsel_period_pclks)
127  {
128  const uint32_t macro_period_ns = calculate_macro_period_ps(vcsel_period_pclks);
129  return ((timeout_period_mclks * macro_period_ns) + 500UL) / 1000UL;
130  }
131  static constexpr uint32_t calculate_timeout_mclks(uint16_t timeout_period_us, uint8_t vcsel_period_pclks)
132  {
133  const uint32_t macro_period_ns = calculate_macro_period_ps(vcsel_period_pclks);
134  return ((timeout_period_us * 1000UL) + (macro_period_ns / 2)) / macro_period_ns;
135  }
136  };
