32 // Device driver guidelines:
33 // - Template on MANAGER, subclass of I2CDevice
34 // - define types aliases PARENT, PROXY and FUTURE
35 // - constructor shall pass one of i2c::Mode i2c::I2C_STANDARD or i2c::I2C_FAST
36 // to inherited constructor, in order tos pecify the BEST mode supported by this driver
37 // - Define FUTURE subclass (inside device class) for every future requiring input (constant, or user-provided)
38 // naming convention: MethodNameFuture
39 // - FUTURE subclass shall have explicit constructor with mandatory input arguments (no default)
40 // - each async API method returns int (error code) and takes a PROXY to specific future as unique argument
41 // - each async API shall have a matching sync API with same name but different prototype:
42 // direct input arguments, reference to output arguments, bool return (true if OK)
43 // each sync API calls the matching async API after creating the proper Future on the stack
44 // then it calls await() or get() on the future before returning.

256  template<uint8_t SIZE_>
257  class SetRamFuture : public FUTURE<void, containers::array<uint8_t, SIZE_ + 1>>
258  {
259  using PARENT = FUTURE<void, containers::array<uint8_t, SIZE_ + 1>>;
260  public:
261  SetRamFuture() = default;
262  explicit SetRamFuture(uint8_t address, const uint8_t (&data)[SIZE_])
263  {
264  static_assert(SIZE_ <= RAM_SIZE, "SIZE_ template paraneter must be less than RAM_SIZE!");
265  typename PARENT::IN input;
266  input[0] = static_cast<uint8_t>(address + RAM_START);
267  input.set(uint8_t(1), data);
268  this->reset_input_(input);
269  }
270  SetRamFuture(SetRamFuture<SIZE_>&&) = default;
271  SetRamFuture& operator=(SetRamFuture<SIZE_>&&) = default;
272 
274  bool is_input_valid() const
275  {
276  return (this->get_input()[0] + SIZE_ <= RAM_END);
277  }
279  };
