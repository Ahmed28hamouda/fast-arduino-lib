127  bool get_datetime(tm& datetime)
128  {
129  // send register address to read from (0)
130  // read datetime at address 0
131  if (write(DEVICE_ADDRESS, TIME_ADDRESS, i2c::BusConditions::START_NO_STOP) == i2c::Status::OK
132  && read(DEVICE_ADDRESS, datetime, i2c::BusConditions::REPEAT_START_STOP) == i2c::Status::OK)
133  {
134  // convert DS1307 output (BCD) to integer type
135  datetime.tm_sec = utils::bcd_to_binary(datetime.tm_sec);

137  datetime.tm_hour = utils::bcd_to_binary(datetime.tm_hour);
138  datetime.tm_mday = utils::bcd_to_binary(datetime.tm_mday);
139  datetime.tm_mon = utils::bcd_to_binary(datetime.tm_mon);
