120  bool get_datetime(tm& datetime)
121  {
122  // send register address to read from (0)
123  // read datetime at address 0
124  if (write(DEVICE_ADDRESS, TIME_ADDRESS, i2c::BusConditions::START_NO_STOP) == i2c::Status::OK
125  && read(DEVICE_ADDRESS, datetime, i2c::BusConditions::REPEAT_START_STOP) == i2c::Status::OK)
