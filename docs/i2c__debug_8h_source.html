125  void trace(streams::ostream& out)

136  void operator()(i2c::DebugStatus status, uint8_t data)
137  {
138  if (index_ >= SIZE) return;
139  switch (status)
140  {
141  case i2c::DebugStatus::START:
142  case i2c::DebugStatus::REPEAT_START:
143  case i2c::DebugStatus::STOP:
144  case i2c::DebugStatus::SLAW:
145  case i2c::DebugStatus::SLAR:
146  case i2c::DebugStatus::SEND:
147  case i2c::DebugStatus::RECV:
148  case i2c::DebugStatus::RECV_LAST:
149  if (uint8_t(debug_) & uint8_t(DEBUG::DEBUG_STEPS))
150  {
151  status_[index_] = status;
152  data_[index_++] = data;
153  }
154  break;
155 
156  case i2c::DebugStatus::SEND_OK:
157  if (uint8_t(debug_) & uint8_t(DEBUG::DEBUG_SEND_OK))
158  {
159  status_[index_] = status;
160  data_[index_++] = data;
161  }
162  break;
163 
164  case i2c::DebugStatus::SEND_ERROR:
165  if (uint8_t(debug_) & uint8_t(DEBUG::DEBUG_SEND_ERR))
166  {
167  status_[index_] = status;
168  data_[index_++] = data;
169  }
170  break;
171 
172  case i2c::DebugStatus::RECV_OK:
173  if (uint8_t(debug_) & uint8_t(DEBUG::DEBUG_RECV_OK))
174  {
175  status_[index_] = status;
176  data_[index_++] = data;
177  }
178  break;
179 
180  case i2c::DebugStatus::RECV_ERROR:
181  if (uint8_t(debug_) & uint8_t(DEBUG::DEBUG_RECV_ERR))
182  {
183  status_[index_] = status;
184  data_[index_++] = data;
185  }
186  break;
187  }
188  }
190

192  i2c::DebugStatus status_[SIZE];
193  uint8_t data_[SIZE];
194  uint8_t index_ = 0;
