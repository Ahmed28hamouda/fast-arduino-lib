110  int read(uint8_t address, uint8_t* data, uint8_t size, BusConditions conditions = BusConditions::START_STOP)
111  {
112  bool ok = true;
113  if (uint8_t(conditions) & Conditions::START)
114  ok = ((uint8_t(conditions) & Conditions::REPEAT) ? manager_.repeat_start() : manager_.start())
115  && manager_.send_slar(address);
116  while (ok && (--size != 0)) ok = manager_.receive_data(*data++);
117  if (uint8_t(conditions) & Conditions::STOP)
118  {
119  ok = ok && manager_.receive_data(*data++, true);
120  manager_.stop();
121  }
122  return manager_.status();
123  }
