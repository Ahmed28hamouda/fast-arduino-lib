246  int launch_commands(
247  PROXY<ABSTRACT_FUTURE> proxy, std::initializer_list<I2CLightCommand> commands)
248  {
249  uint8_t num_commands = commands.size();
250  if (num_commands == 0) return errors::EINVAL;
251  {
252  // Truly asynchronous mode (ATmega only)
253  // The whole code block in launch_commands() must be synchronized
254  UNUSED auto outer_sync = DisableInterrupts<MANAGER_TRAIT::IS_ASYNC>{};
255  uint8_t max_read;
256  uint8_t max_write;
257  {
258  // Synchronous mode
259  // Only the next few method calls shall be synchronized
260  UNUSED auto inner_sync = DisableInterrupts<!MANAGER_TRAIT::IS_ASYNC>{};
261  // pre-conditions (must be synchronized)
262  if (!handler_.ensure_num_commands_(num_commands)) return errors::EAGAIN;
263  ABSTRACT_FUTURE& future = resolve(proxy);
264  max_read = future.get_future_value_size_();
265  max_write = future.get_storage_value_size_();
266  }
267 
268  // That check is normally usefull only in debug mode
269  if (MANAGER_TRAIT::IS_DEBUG && (!check_commands(max_write, max_read, commands)))
270  return errors::EINVAL;
271 
272  // Now push each command to the I2C Manager
273  int error = 0;
274  for (I2CLightCommand command : commands)
275  {
276  // update command.byte_count if 0
277  command.update_byte_count(max_read, max_write);
278  // force future finish for last command in transaction
279  --num_commands;
280  if (num_commands == 0)
281  command.type().add_flags(auto_stop_flags_);
282  // Note: on ATtiny, this method blocks until I2C command is finished!
283  if (!handler_.push_command_(command, device_, proxy))
284  {
285  error = errors::EPROTO;
286  break;
287  }
288  }
289  // Notify handler that transaction is complete
290  handler_.last_command_pushed_();
291  return error;
292  }
293  }

319  static bool check_commands(
320  uint8_t max_write, uint8_t max_read, const std::initializer_list<I2CLightCommand>& commands)
321  {
322  // Limit total number of bytes read or written in a transaction to 255
323  uint8_t total_read = 0;
324  uint8_t total_write = 0;
325  for (const I2CLightCommand& command : commands)
326  {
327  // Count number of bytes read and written
328  if (command.type().is_write())
329  total_write += (command.byte_count() ? command.byte_count() : max_write);
330  else
331  total_read += (command.byte_count() ? command.byte_count() : max_read);
332  }
333  // check sum of read commands byte_count matches future output size
334  // check sum of write commands byte_count matches future input size
335  return (total_write == max_write) && (total_read == max_read);
336  }

341  };
