235  int launch_commands(
236  PROXY<ABSTRACT_FUTURE> proxy, std::initializer_list<I2CLightCommand> commands)
237  {
238  uint8_t num_commands = commands.size();
239  if (num_commands == 0) return errors::EINVAL;
240  OUTER_SYNCHRONIZED
241  {
242  uint8_t max_read;
243  uint8_t max_write;
244  INNER_SYNCHRONIZED
245  {
246  // pre-conditions (must be synchronized)
247  if (!handler_.ensure_num_commands_(num_commands)) return errors::EAGAIN;
248  ABSTRACT_FUTURE& future = resolve(proxy);
249  max_read = future.get_future_value_size_();
250  max_write = future.get_storage_value_size_();
251  }
252 
253  // That check is normally usefull only in debug mode
254  if (MANAGER_TRAIT::IS_DEBUG)
255  {
256  // Limit total number of bytes read or written in a transaction to 255
257  uint8_t total_read = 0;
258  uint8_t total_write = 0;
259  for (const I2CLightCommand& command : commands)
260  {
261  // Count number of bytes read and written
262  if (command.type().is_write())
263  total_write += (command.byte_count() ? command.byte_count() : max_write);
264  else
265  total_read += (command.byte_count() ? command.byte_count() : max_read);
266  }
267  // check sum of read commands byte_count matches future output size
268  // check sum of write commands byte_count matches future input size
269  if ((total_write != max_write) || (total_read != max_read)) return errors::EINVAL;
270  }
271 
272  // Now push each command to the I2C Manager
273  int error = 0;
274  for (I2CLightCommand command : commands)
275  {
276  // update command.byte_count if 0
277  command.update_byte_count(max_read, max_write);
278  // force future finish for last command in transaction
279  if (--num_commands == 0)
280  command.type().add_flags(I2CCommandType::flags(auto_stop_, true, true));
281  // Note: on ATtiny, this method blocks until I2C command is finished!
282  if (!handler_.push_command_(command, device_, proxy))
283  {
284  error = errors::EPROTO;
285  break;
286  }
287  }
288  // Notify handler that transaction is complete
289  handler_.last_command_pushed_();
290  return error;
291  }
292  }
