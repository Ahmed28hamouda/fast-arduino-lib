383  bool get(Sensor3D& fields)
384  {
385  if (!PARENT::get(fields)) return false;
386  utils::swap_bytes(fields.x);
387  utils::swap_bytes(fields.y);
388  utils::swap_bytes(fields.z);
389  // HMC5883L registers are in order X,Z,Y while Sensor3D is X,Y,Z
390  int16_t temp = fields.y;
391  fields.y = fields.z;
392  fields.z = temp;
393  return true;
394  }
396  };
