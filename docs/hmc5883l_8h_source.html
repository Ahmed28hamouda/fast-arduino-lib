227  bool begin(OperatingMode mode = OperatingMode::SINGLE, Gain gain = Gain::GAIN_1_3GA,
228  DataOutput rate = DataOutput::RATE_15HZ, SamplesAveraged samples = SamplesAveraged::ONE_SAMPLE,
229  MeasurementMode measurement = MeasurementMode::NORMAL)

273  bool magnetic_fields(Sensor3D& fields)
274  {
275  using i2c::Status::OK;
276  if (this->write(DEVICE_ADDRESS, OUTPUT_REG_1, BUSCOND::START_NO_STOP) == OK
277  && this->read(DEVICE_ADDRESS, fields, BUSCOND::REPEAT_START_STOP) == OK)
278  {
279  utils::swap_bytes(fields.x);
280  utils::swap_bytes(fields.y);
281  utils::swap_bytes(fields.z);
282  return true;
283  }
284  else

336  static constexpr uint16_t GAIN(Gain gain)
337  {
338  if (gain == Gain::GAIN_0_88GA) return 1370;
339  if (gain == Gain::GAIN_1_3GA) return 1090;
340  if (gain == Gain::GAIN_1_9GA) return 820;

343  if (gain == Gain::GAIN_4_7GA) return 390;
