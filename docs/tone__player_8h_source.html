221  // Flags meaning
222  static constexpr uint8_t TONE = 0x00;
223  static constexpr uint8_t NONE = 0x01;

233  static constexpr uint32_t period(Tone tone)
234  {
235  return ONE_SECOND / 2 / uint16_t(tone);
236  }
237  static constexpr PRESCALER prescaler(Tone tone)
238  {
239  return (tone > Tone::SILENCE ? CALC::CTC_prescaler(period(tone)) : PRESCALER::NO_PRESCALING);
240  }
241  static constexpr COUNTER counter(Tone tone)
242  {
243  return (tone > Tone::SILENCE ? CALC::CTC_counter(prescaler(tone), period(tone)) : 0);
244  }

246  {

353  }
354

365  {
366  return stop_current_();
367  }
368 
372  bool is_finished() const
373  {
374  return (loader_ == nullptr);
375  }
376 
377  private:
378  static constexpr const uint16_t INTERTONE_DELAY_MS = 20;
379 
380  using LOAD_TONE = const TONE_PLAY* (*) (const TONE_PLAY* address, TONE_PLAY& holder);
381 
382  void prepare_(const TONE_PLAY* melody, LOAD_TONE load_tone)
383  {
384  loader_ = load_tone;
385  current_play_ = melody;
386  repeat_play_ = nullptr;
387  repeat_times_ = 0;
388  }
389 
390  void reset_()
391  {
392  loader_ = nullptr;
393  current_play_ = nullptr;
394  repeat_play_ = nullptr;
395  repeat_times_ = 0;
396  }
397 
398  uint16_t start_next_()
399  {
400  if (loader_ == nullptr) return 0;
401 
402  uint16_t delay = 0;
403  TONE_PLAY holder;
404  const TONE_PLAY* current = loader_(current_play_, holder);
405  if (current->is_end())
406  {
407  reset_();
408  return 0;
409  }
410  if (current->is_repeat_start())
411  {
412  repeat_play_ = current_play_;
413  repeat_times_ = -1;
414  no_delay_ = true;
415  }
416  else if (current->is_repeat_end())
417  {
418  if (repeat_play_)
419  {
420  if (repeat_times_ == -1) repeat_times_ = current->repeat_count();
421  if (repeat_times_--)
422  current_play_ = repeat_play_;
423  else
424  repeat_play_ = nullptr;
425  }
426  no_delay_ = true;
427  }
428  else
429  {
430  if (current->is_tone()) current->generate_tone(generator_);
431  delay = current->duration();
432  no_delay_ = false;
433  }
434  ++current_play_;
435  return delay;
436  }
437 
438  uint16_t stop_current_()
439  {
440  if (no_delay_)
441  return 0;
442  generator_.stop_tone();
443  return INTERTONE_DELAY_MS;
444  }
445 
446  static const TONE_PLAY* load_sram(const TONE_PLAY* address, TONE_PLAY& holder UNUSED)
447  {
448  return address;
449  }
450  static const TONE_PLAY* load_eeprom(const TONE_PLAY* address, TONE_PLAY& holder)
451  {
452  eeprom::EEPROM::read(address, holder);
453  return &holder;
454  }
455  static const TONE_PLAY* load_flash(const TONE_PLAY* address, TONE_PLAY& holder)
456  {
457  flash::read_flash(address, holder);
458  return &holder;
459  }
460 
461  GENERATOR& generator_;
462  LOAD_TONE loader_;
463  const TONE_PLAY* current_play_;
464  const TONE_PLAY* repeat_play_;
465  int8_t repeat_times_;
466  bool no_delay_;
467  };

596  stop_ = true;
597  }
598  }

748  {
749  uint16_t delay;
750  Status next;
751  if (status_ == Status::PLAYING_NOTE)
752  {
753  delay = this->stop_current_note();
754  next = Status::PLAYING_INTERNOTE;
755  }
756  else
757  {
758  delay = this->start_next_note();
759  next = Status::PLAYING_NOTE;
760  }
761 
762  if (this->is_finished())
763  status_ = Status::NOT_STARTED;
764  else
765  {
766  next_time_ = rtt_millis + delay;
767  status_ = next;
768  }
769  }
