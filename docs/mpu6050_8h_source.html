857  class FifoPopFuture : public FUTURE<T, uint8_t>

866  bool get(T& result)

869  // We need to swap bytes of all words (big-endian to little-endian)
870  uint16_t* temp = reinterpret_cast<uint16_t*>(&result);

872  while (size--) utils::swap_bytes(*temp++);
873  return true;
874  }
