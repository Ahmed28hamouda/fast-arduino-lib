837  template<typename T>
838  class FifoPopFuture : public FUTURE<T, uint8_t>
839  {
840  using PARENT = FUTURE<T, uint8_t>;
841  public:
842  FifoPopFuture() : PARENT{FIFO_R_W} {}

846  bool get(T& result)
847  {
848  if (!PARENT::get(result)) return false;
849  // We need to swap bytes of all words (big-endian to little-endian)
850  uint16_t* temp = reinterpret_cast<uint16_t*>(&result);
851  uint8_t size = sizeof(T) / 2;
852  while (size--) utils::swap_bytes(*temp++);
853  return true;
854  }
855  };
