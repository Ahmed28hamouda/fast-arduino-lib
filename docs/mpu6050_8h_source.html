288  bool begin( FIFOEnable fifo_enable,
289  INTEnable int_enable,

291  GyroRange gyro_range = GyroRange::RANGE_250,
292  AccelRange accel_range = AccelRange::RANGE_2G,

294  ClockSelect clock_select = ClockSelect::INTERNAL_8MHZ)
295  {
296  using i2c::Status::OK;

298  if (utils::as_uint8_t(fifo_enable) == 0)
299  return begin(gyro_range, accel_range, low_pass_filter, clock_select);
300  int_enable.reserved1 = int_enable.reserved2 = 0;
301  PowerManagement power;

303  return this->write(DEVICE_ADDRESS, CONFIG, BUSCOND::START_NO_STOP) == OK

309  && this->write(DEVICE_ADDRESS, sample_rate_divider, BUSCOND::NO_START_NO_STOP) == OK
310 
311  && this->write(DEVICE_ADDRESS, FIFO_EN, BUSCOND::REPEAT_START_NO_STOP) == OK

588  bool fifo_pop(uint8_t* buffer, uint8_t size, bool wait, bool yield)
589  {
590  using i2c::Status::OK;
591  while (fifo_count() < size)
592  {
593  if (!wait) return false;
594  if (yield) time::yield();
595  }
596 
597  if (this->write(DEVICE_ADDRESS, FIFO_R_W, BUSCOND::START_NO_STOP) == OK
598  && this->read(DEVICE_ADDRESS, buffer, size, BUSCOND::REPEAT_START_STOP) == OK)
599  {
600  // Swap all 2-bytes words
601  uint16_t* temp = (uint16_t*) buffer;
602  size /= 2;
603  while (size--) utils::swap_bytes(*temp++);
604  return true;
605  }
606  return false;
607  }
