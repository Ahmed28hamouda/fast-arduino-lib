592  template<board::Timer NTIMER_> class AbstractSonar
593  {
594  public:
596  using RTT = timer::RTT<NTIMER_>;
597 
607  bool ready() const
608  {
609  return status_ == READY;
610  }
611 
622  uint16_t latest_echo_us() const
623  {
624  synchronized
625  {
626  if (status_ == READY)
627  return echo_time_();
628  else
629  return 0;
630  }
631  }
632 
633  private:
634  using RAW_TIME = typename RTT::RAW_TIME;
635 
636  protected:
638  explicit AbstractSonar(const RTT& rtt)
639  : rtt_{rtt}, status_{UNKNOWN}, timeout_time_ms_{},
640  echo_start_{RAW_TIME::EMPTY_TIME}, echo_end_{RAW_TIME::EMPTY_TIME}
641  {}
642 
643  uint16_t async_echo_us(uint16_t timeout_ms)
644  {
645  uint32_t now = rtt_.millis();
646  now += timeout_ms;
647  // Wait for echo signal start
648  while (status_ != READY)
649  if (rtt_.millis() >= now)
650  {
651  synchronized
652  {
653  status_ = READY;
654  echo_start_ = echo_end_ = RAW_TIME::EMPTY_TIME;
655  }
656  return 0;
657  }
658  return echo_time_();
659  }
660 
661  template<board::DigitalPin ECHO>
662  uint16_t blocking_echo_us(typename gpio::FastPinType<ECHO>::TYPE& echo, uint16_t timeout_ms)
663  {
664  uint32_t now = rtt_.millis();
665  now += timeout_ms;
666  while (!echo.value())
667  if (rtt_.millis() >= now) return 0;
668  synchronized
669  {
670  status_ = ECHO_STARTED;
671  echo_start_ = rtt_.raw_time();
672  }
673  // Wait for echo signal end
674  while (echo.value())
675  if (rtt_.millis() >= now) return 0;
676  synchronized
677  {
678  status_ = READY;
679  echo_end_ = rtt_.raw_time();
680  return echo_time_();
681  }
682  }
683 
684  void trigger_sent(uint16_t timeout_ms)
685  {
686  synchronized
687  {
688  status_ = TRIGGERED;
689  timeout_time_ms_ = rtt_.millis_() + timeout_ms;
690  }
691  }
692 
693  bool pulse_edge(bool rising)
694  {
695  if (rising && (status_ == TRIGGERED))
696  {
697  status_ = ECHO_STARTED;
698  echo_start_ = rtt_.raw_time_();
699  return false;
700  }
701  if ((!rising) && (status_ == ECHO_STARTED))
702  {
703  status_ = READY;
704  echo_end_ = rtt_.raw_time_();
705  return true;
706  }
707  return false;
708  }
709 
710  bool rtt_time_changed()
711  {
712  if ((status_ != READY) && (rtt_.millis_() >= timeout_time_ms_))
713  {
714  status_ = READY;
715  echo_start_ = echo_end_ = RAW_TIME::EMPTY_TIME;
716  return true;
717  }
718  return false;
719  }
721 
722  private:
723  uint16_t echo_time_() const
724  {
725  return uint16_t((echo_end_.as_real_time() - echo_start_.as_real_time()).total_micros());
726  }
727 
728  const RTT& rtt_;
729 
730  static constexpr const uint8_t UNKNOWN = 0x00;
731  static constexpr const uint8_t TRIGGERED = 0x10;
732  static constexpr const uint8_t ECHO_STARTED = 0x11;
733  static constexpr const uint8_t READY = 0x20;
734 
735  volatile uint8_t status_;
736  uint32_t timeout_time_ms_;
737 
738  RAW_TIME echo_start_;
739  RAW_TIME echo_end_;
740  };
