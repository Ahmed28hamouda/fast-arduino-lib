361  {

394  bool set_future_value_(uint8_t chunk)
395  {
396  // Check this future is waiting for data
397  if (status_ != FutureStatus::NOT_READY)
398  return false;
399  // Update Future value chunk
400  *output_current_++ = chunk;
401  // Is that the last chunk?
402  if (--output_size_ == 0)
403  status_ = FutureStatus::READY;
404  return true;
405  }

434  bool set_future_value_(const uint8_t* chunk, uint8_t size)
435  {
436  // Check size does not go beyond expected size
437  if (size > output_size_)
438  {
439  // Store error
440  set_future_error_(errors::EMSGSIZE);
441  return false;
442  }
443  while (size--)
444  {
445  if (!set_future_value_(*chunk++)) return false;
446  }
447  return true;
448  }

497  bool set_future_error_(int error)
498  {

522  // This method is called by subclass to check if input is replaceable,
523  // i.e. it has not been read yet

529  // This method is called by subclass in their move constructor and assignment operator
530  void move_(AbstractFuture&& that, uint8_t full_output_size, uint8_t full_input_size)
531  {
532  // Now copy all attributes from rhs (output_data_ was already initialized when this was constructed)
533  const FutureStatus status = that.status_;
534  status_ = status;
535  // Make rhs Future invalid
536  if (status == FutureStatus::NOT_READY)
537  that.status_ = FutureStatus::INVALID;
538 
539  error_ = that.error_;
540  const uint8_t output_size = that.output_size_;
541  // Calculate data pointer attribute for next set value calls
542  output_current_ = output_data_ + full_output_size - output_size;
543  output_size_ = output_size;
544 
545  const uint8_t input_size = that.input_size_;
546  // Calculate data pointer attribute for next set value calls
547  input_current_ = input_data_ + full_input_size - input_size;
548  input_size_ = input_size;
549  }
551 
552  private:
553  volatile FutureStatus status_ = FutureStatus::NOT_READY;
554  int error_ = 0;
555 
556  uint8_t* output_data_ = nullptr;
557  uint8_t* output_current_ = nullptr;
558  uint8_t output_size_ = 0;
559 
560  uint8_t* input_data_ = nullptr;
561  uint8_t* input_current_ = nullptr;
562  uint8_t input_size_ = 0;
563  };
