336  template<typename T> bool write_register(uint8_t address, T value)
337  {
338  using i2c::Status::OK;
339  return this->write(device_, address, BUSCOND::START_NO_STOP) == OK
340  && this->write(device_, value, BUSCOND::NO_START_STOP) == OK;
341  }
342 
343  template<typename T> bool read_register(uint8_t address, T& value)
344  {
345  using i2c::Status::OK;
346  return this->write(device_, address, BUSCOND::START_NO_STOP) == OK
347  && this->read(device_, value, BUSCOND::REPEAT_START_STOP) == OK;
348  }
