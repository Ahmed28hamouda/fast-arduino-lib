308  template<typename T> bool write_register(uint8_t address, T value)
309  {
310  using i2c::Status::OK;
311  return this->write(device_, address, BUSCOND::START_NO_STOP) == OK
312  && this->write(device_, value, BUSCOND::NO_START_STOP) == OK;
313  }
314 
315  template<typename T> bool read_register(uint8_t address, T& value)
316  {
317  using i2c::Status::OK;
318  return this->write(device_, address, BUSCOND::START_NO_STOP) == OK
319  && this->read(device_, value, BUSCOND::REPEAT_START_STOP) == OK;
320  }
