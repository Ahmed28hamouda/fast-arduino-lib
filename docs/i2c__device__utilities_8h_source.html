430  template<typename MANAGER> class I2CFuturesGroup :
431  public AbstractI2CFuturesGroup<MANAGER>
432  {
433  using PARENT = AbstractI2CFuturesGroup<MANAGER>;
434  using MANAGER_TRAIT = I2CManager_trait<MANAGER>;
435  using STATUS_LISTENER = typename PARENT::STATUS_LISTENER;
436  using ABSTRACT_FUTURE = typename PARENT::ABSTRACT_FUTURE;
437 
438  protected:
447  I2CFuturesGroup(
448  ABSTRACT_FUTURE** futures, uint8_t size, STATUS_LISTENER* status_listener = nullptr)
449  : PARENT{status_listener}, futures_{futures}, size_{size} {}
450 
465  bool start(typename PARENT::DEVICE& device)
466  {
467  PARENT::set_device(device);
468  if (MANAGER_TRAIT::IS_ASYNC)
469  {
470  return next_future();
471  }
472  else
473  {
474  // In sync mode, we replace recursive calls (generated by on_status_change()) by a loop
475  while (index_ != size_)
476  {
477  if (!next_future()) return false;
478  }
479  return true;
480  }
481  }
482 
484  void on_status_change(const ABSTRACT_FUTURE& future, future::FutureStatus status) final
485  {
486  PARENT::on_status_change(future, status);
487  // In sync mode, we must avoid recursive calls generated by on_status_change()!
488  if (MANAGER_TRAIT::IS_ASYNC)
489  {
490  // First check that current future was executed successfully
491  if (status == future::FutureStatus::READY)
492  next_future();
493  }
494  }
496 
497  private:
498  bool next_future()
499  {
500  if (index_ == size_)
501  // Future is finished
502  return false;
503 
504  ABSTRACT_FUTURE& future = *futures_[index_++];
505  // Check if future has read, write or both
506  const bool stop = (index_ == size_);
507  const bool read = future.get_future_value_size_();
508  const bool write = future.get_storage_value_size_();
509  int error = 0;
510  if (read && write)
511  {
512  error = PARENT::launch_commands(future, {PARENT::write(), PARENT::read(0, false, stop)});
513  }
514  else if (read)
515  {
516  error = PARENT::launch_commands(future, {PARENT::read(0, false, stop)});
517  }
518  else if (write)
519  {
520  error = PARENT::launch_commands(future, {PARENT::write(0, false, stop)});
521  }
522  else
523  {
524  //FIXME we consider that any other future is an I2CFuturesgroup, which might not always be correct!
525  I2CFuturesGroup& group = static_cast<I2CFuturesGroup&>(future);
526  if (!group.start(PARENT::device()))
527  error = errors::EILSEQ;
528  }
529  return PARENT::check_error(error);
530  }
531 
532  ABSTRACT_FUTURE** futures_;
533  uint8_t size_;
534  uint8_t index_ = 0;
535  };

556  using ABSTRACT_FUTURE = typename PARENT::ABSTRACT_FUTURE;
557  using F = WriteRegisterFuture<MANAGER, uint8_t, true>;
558  using CONTENT = WriteContent<uint8_t, true>;
559  static constexpr uint8_t FUTURE_SIZE = F::IN_SIZE;
560 
561  public:
570  I2CSameFutureGroup(uint16_t address, uint8_t size, STATUS_LISTENER* status_listener = nullptr)
571  : PARENT{status_listener}, address_{address}, size_{size}
572  {
573  PARENT::init({&future_}, size / FUTURE_SIZE);
574  }

590  bool start(typename PARENT::DEVICE& device)
591  {
592  PARENT::set_device(device);
593  if (MANAGER_TRAIT::IS_ASYNC)
594  {
595  return next_future();
596  }
597  else
598  {
599  // In sync mode, we replace recursive calls (generated by on_status_change()) by a loop
600  while (size_)
601  {
602  if (!next_future()) return false;
603  }
604  return true;
605  }
606  }
607 
608  private:
609  bool next_future()
610  {
611  if (size_ == 0)
612  // Future is finished already
613  return false;
614 
615  uint8_t reg = next_byte();
616  uint8_t val = next_byte();
617  const bool stop = (size_ == 0);
618  future_.reset_(CONTENT{reg, val});
619  return PARENT::check_error(
620  PARENT::launch_commands(future_, {PARENT::write(0, false, stop)}));
621  }
622 
623  // Get the next byte, from the flash
624  uint8_t next_byte()
625  {
626  uint8_t data = 0;
627  --size_;
628  return flash::read_flash(address_++, data);
629  }
630 
631  void on_status_change(const ABSTRACT_FUTURE& future, future::FutureStatus status) final
632  {
633  PARENT::on_status_change(future, status);
634  // In sync mode, we must avoid recursive calls generated by on_status_change()!
635  if (MANAGER_TRAIT::IS_ASYNC)
636  {
637  // First check that current future was executed successfully
638  if (status == future::FutureStatus::READY)
639  next_future();
640  }
641  }
642 
643  // Address of flah mempry holding information about bytes to write
644  uint16_t address_;
645  uint8_t size_;
646  // The future reused for all writes
647  F future_{0, 0};

767  ProcessAction process_action()
768  {
769  while ((action_ = next_byte()) == actions::LOOP)
770  {
771  // Store loop address for later use and skip to next action
772  loop_ = address_;
773  }
774  if (action_ == actions::END)
775  {
776  // Future is finished

781  if (action_ == actions::INCLUDE) return ProcessAction::INCLUDE;
782  if (actions::is_read(action_)) return ProcessAction::READ;
783  if (actions::is_write(action_)) return ProcessAction::WRITE;
784

786  PARENT::check_error(errors::EILSEQ);
787  return ProcessAction::DONE;
788  }
