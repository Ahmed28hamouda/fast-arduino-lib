196  constexpr int16_t map_raw_to_physical(int16_t value, UnitPrefix prefix, int16_t range, uint8_t precision_bits)
197  {
198  // Here we approximate the calculation by using 2^n instead of (2^n - 1) as input range
199  const int8_t prefix_value = int8_t(prefix);
200  if (prefix_value > 0)
201  return (int32_t(value) * int32_t(range) / int32_t(power_of_10(prefix_value))) >> precision_bits;
202  else
203  return (int32_t(value) * int32_t(range) * int32_t(power_of_10(prefix_value))) >> precision_bits;
204  }

207  // This is intermediate function used by map_physical_to_raw() but without overflow issue (int32 only)
208  constexpr int32_t map_physical_to_raw_(int32_t value, int8_t prefix, int32_t range, uint8_t precision_bits)
209  {
210  // Here we approximate the calculation by using 2^n instead of (2^n - 1) as input range
211  if (prefix >= 0)
212  return (value << precision_bits) * int32_t(power_of_10(prefix)) / range;
213  else
214  return (value << precision_bits) / int32_t(power_of_10(prefix)) / range;
215  }
