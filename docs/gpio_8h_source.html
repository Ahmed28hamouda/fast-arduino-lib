135  void set_mode(PinMode mode, bool value = false) INLINE
136  {
137  if (mode == PinMode::OUTPUT)
138  TRAIT::DDR |= bits::BV8(BIT);
139  else
140  TRAIT::DDR &= bits::CBV8(BIT);
141  if (value || (mode == PinMode::INPUT_PULLUP))
142  TRAIT::PORT |= bits::BV8(BIT);
143  else
144  TRAIT::PORT &= bits::CBV8(BIT);
145  }
